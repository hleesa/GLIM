#include <string>
#include <vector>

using namespace std;

int solution(vector<vector<int>> board, int k) {
    int sum = 0;
    for(size_t i = 0; i < board.size(); ++i){
        for(size_t j = 0; j < board.front().size(); ++j){
            if(i + j <= k){
                sum += board[i][j];
            }
        }
    }
    return sum;
}
