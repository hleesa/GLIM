#include <string>
#include <vector>
#include <stack>
#include <sstream>

using namespace std;

int solution(string s) {
    stringstream ss(s);
    string token;
    int sum = 0;
    int lastNum = 0;
    
    while(ss >> token){
        if(token == "Z"){
            sum -= lastNum;
        }
        else {
            lastNum = stoi(token);
            sum += lastNum;
        }
    }
    
    return sum;
}
